#@title 3.Benchmark Model
import pandas as pd
from sklearn.feature_selection import RFE
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import GradientBoostingClassifier, RandomForestClassifier, StackingClassifier
from sklearn.svm import SVC
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import accuracy_score, f1_score, confusion_matrix, classification_report

dataset = data.copy()

# Feature set
X = data.drop(columns=['price_range'])

# Target variable
y = data['price_range']

# Split dataset into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)
columns=X_train.columns

# Standardizing the features
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

# RandomForest
model = RandomForestClassifier(n_estimators=100, random_state=42)

# Training the model with selected features
model.fit(X_train, y_train)

# Get feature importances
importances = model.feature_importances_

# Convert the importances into one-dimensional 1darray with corresponding df column names as axis labels
f_importances = pd.Series(importances, columns)

# Sort the array in ascending order of the importances
f_importances = f_importances.sort_values()

# Make the bar Plot from f_importances
f_importances.plot(x='Features', y='Importance', kind='barh', figsize=(9,6), legend=False)
plt.title('Feature Importances from Random Forest')
plt.xlabel('Relative Importance')
plt.show()

# Make predictions
y_pred = model.predict(X_test)

# Calculate the accuracy
print("Model Accuracy:", accuracy_score(y_test, y_pred))

# Calculate the F1-score
f1 = f1_score(y_test, y_pred, average='weighted')
print(f"F1-Score: {f1}")

# Calculate the confusion matrix
cm = confusion_matrix(y_test, y_pred)
print(f"Confusion Matrix: \n{cm}")

print("\nClassification Report:\n", classification_report(y_test, y_pred))

#@title 4.Feature Engineering
# Adding new features to the dataset

# 'px_area' is created by multiplying pixel resolution height by pixel resolution width
data['px_area'] = data['px_height'] * data['px_width']

# 'sc_area' is created by multiplying screen height by screen width
data['sc_area'] = data['sc_h'] * data['sc_w']

# 'battery_per_weight' is created by dividing battery power by mobile weight
data['battery_per_weight'] = data['battery_power'] / data['mobile_wt']

# 'memory_per_core' is created by dividing internal memory by the number of cores of the processor
data['memory_per_core'] = data['int_memory'] / data['n_cores']

# 'camera_quality' is created by summing the front camera mega pixels and primary camera mega pixels
data['camera_quality'] = data['fc'] + data['pc']

# 'connectivity' is created by summing the features that define connectivity capabilities (4G, 3G, wifi, bluetooth, dual sim)
data['connectivity'] = data['four_g'] + data['three_g'] + data['wifi'] + data['blue'] + data['dual_sim']

# Ratio of screen area to mobile weight to give an estimate of screen to body ratio. Avoid division by zero.
data['screen_to_body_ratio'] = data['sc_area'] / (data['mobile_wt'] + 1e-3)

# Avoid division by zero by adding a small constant.
data['battery_to_ram'] = data['battery_power'] / (data['ram'] + 1e-3)

# Pixel density is the pixel area per screen area. Avoid division by zero.
data['px_density'] = data['px_area'] / (data['sc_area'] + 1e-3)

# Avoid division by zero by adding a small constant.
data['camera_to_ram'] = data['camera_quality'] / (data['ram'] + 1e-3)

# Replace infinity values with NaN
data.replace([np.inf, -np.inf], np.nan, inplace=True)

# Handle NaN values (you can fill them with mean, median, or drop them)
data.fillna(data.mean(), inplace=True)

#@title 5.Dimension Reduction with PCA
from sklearn.decomposition import PCA
# Feature set
X = data.drop(columns=['price_range'])

# Target variable
y = data['price_range']

# Split dataset into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Standardizing the features
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

# Plot model accuracy against the number of principal components
accuracies = []
for i in range(1, 11):
    # Perform PCA with i components
    pca = PCA(n_components=i)
    X_train_pca = pca.fit_transform(X_train)
    X_test_pca = pca.transform(X_test)

    # Train a model
    model = RandomForestClassifier(n_estimators=100, random_state=42)
    model.fit(X_train_pca, y_train)

    # Predict on the test set and calculate accuracy
    y_pred = model.predict(X_test_pca)
    acc = accuracy_score(y_test, y_pred)
    accuracies.append(acc)

# Plot the model accuracy for each number of components
plt.figure()
plt.plot(range(1, 11), accuracies)
plt.xlabel('Number of Components')
plt.ylabel('Accuracy')
plt.title('Model Accuracy vs. Number of PCA Components')
plt.xticks(range(1, 11))
plt.grid()
plt.show()

#@title 6.Dimension Reduction with LDA
from sklearn.discriminant_analysis import LinearDiscriminantAnalysis as LDA

# Feature set
X = data.drop(columns=['price_range'])

# Target variable
y = data['price_range']

# Split dataset into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Standardizing the features
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

# Perform LDA with optimal number of components
# The number of components for LDA should be <= min(n_classes - 1, n_features)
lda = LDA(n_components=min(len(data['price_range'].unique())-1, X.shape[1]))
X_train_lda = lda.fit_transform(X_train, y_train)
X_test_lda = lda.transform(X_test)

# Analyzing explained_variance_ratio_
explained_var_ratios = lda.explained_variance_ratio_
print("Explained variance ratios of LDA components:", explained_var_ratios)

# Visualize the contribution of features to each discriminant
feature_contributions = lda.scalings_

# Plotting the heatmap
plt.figure(figsize=(10, 5))
sns.heatmap(feature_contributions, annot=True, cmap='viridis', yticklabels=data.drop(columns=['price_range']).columns)
plt.xlabel("LDA Component")
plt.ylabel("Feature")
plt.title("Feature Contributions to LDA Components")
plt.show()

# Initialize and train the model
model = RandomForestClassifier(n_estimators=100, random_state=42)
model.fit(X_train_lda, y_train)

# Predict on the test set
y_pred = model.predict(X_test_lda)

# Evaluate the model
# Calculate the accuracy
print("Model Accuracy:", accuracy_score(y_test, y_pred))

# Calculate the F1-score
f1 = f1_score(y_test, y_pred, average='weighted')
print(f"F1-Score: {f1}")

# Calculate the confusion matrix
cm = confusion_matrix(y_test, y_pred)
print(f"Confusion Matrix: \n{cm}")
print("\nClassification Report:\n", classification_report(y_test, y_pred))

#@title 7.2 Model Enhancement - Stacking with LDA
# Feature set
X = data.drop(columns=['price_range'])

# Target variable
y = data['price_range']

# Split dataset into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Standardizing the features
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

# LDA
# best n_components of LDA is min(len(data['price_range'].unique())-1, X.shape[1]))=2
lda = LDA(n_components=min(len(data['price_range'].unique())-1, X.shape[1]))
X_train_lda = lda.fit_transform(X_train, y_train)
X_test_lda = lda.transform(X_test)

# Training the stacked model with LogisticRegression and SVM
estimators = [('lr', LogisticRegression(max_iter=1000)),
              ('svc', SVC()),
              ('rf', RandomForestClassifier(n_estimators=100, random_state=42)),
              ('gb', GradientBoostingClassifier(n_estimators=100, random_state=42))]
stacking_model = StackingClassifier(estimators=estimators, final_estimator=LogisticRegression(max_iter=1000))
stacking_model.fit(X_train_lda, y_train)

# Make predictions
y_pred = stacking_model.predict(X_test_lda)

# Calculate the accuracy
print("Model Accuracy:", accuracy_score(y_test, y_pred))

# Calculate the confusion matrix
cm = confusion_matrix(y_test, y_pred)
print(f"Confusion Matrix: \n{cm}")

print("\nClassification Report:\n", classification_report(y_test, y_pred))
